# Lecoo GUI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Построить полноценное desktop GUI поверх существующего daemon'а Lecoo Control Center в форке `A-mi13/leeco-control-center`, заполнив пустой crate `gui/`.

**Architecture:** Tauri 2 (Rust backend + React/TS webview), переиспользует существующий `ipc`-крейт оригинала через workspace dependency `../ipc`. Daemon → named pipe → gui-backend (Rust) → Tauri IPC → webview (React). Параллельно дорабатываем daemon в нашем форке для server-side fan curve и других фиксов.

**Tech Stack:** Tauri 2.x, Rust, React 18 + TypeScript, Tailwind CSS, uPlot, i18next, Zustand, Radix UI primitives.

**Spec:** `docs/superpowers/specs/2026-05-22-lecoo-gui-design.md`

pub struct IpcClient {

### Task 1.4: Poller with reconnect loop

**Files:**
- Modify: `gui/src/poller.rs`

- [ ] **Step 1: Implement poller**

```rust
use crate::ipc_client::IpcClient;
use crate::state::{AppState, ConnectionStatus};
use std::sync::Arc;
use std::time::Duration;
use tauri::{AppHandle, Emitter, Manager};

const POLL_INTERVAL_MS: u64 = 1000;
const RECONNECT_DELAY_MS: u64 = 2000;
const GUI_VERSION: (u8, u8) = (0, 1);

pub fn start(app: AppHandle) {
    tokio::spawn(async move {
        run(app).await;
    });
}

async fn run(app: AppHandle) {
    let state = app.state::<Arc<AppState>>().inner().clone();

    loop {
        *state.connection.write() = ConnectionStatus::Connecting;
        let _ = app.emit("connection-status", &*state.connection.read());

        let mut client = match IpcClient::connect() {
            Ok(c) => {
                let daemon_version = c.daemon_version;
                let status = if daemon_version.0 != GUI_VERSION.0 {
                    ConnectionStatus::Mismatch { daemon: daemon_version, gui: GUI_VERSION }
                } else {
                    ConnectionStatus::Connected { daemon_version }
                };
                *state.connection.write() = status.clone();
                let _ = app.emit("connection-status", &status);
                c
            }
            Err(e) => {
                *state.connection.write() = ConnectionStatus::Error(e.to_string());
                let _ = app.emit("connection-status", &*state.connection.read());
                tokio::time::sleep(Duration::from_millis(RECONNECT_DELAY_MS)).await;
                continue;
            }
        };

        loop {
            match client.fetch_telemetry() {
                Ok(t) => {
                    *state.last_telemetry.write() = Some(t.clone());
                    let _ = app.emit("telemetry", &t);
                }
                Err(e) => {
                    tracing::warn!("poll failed: {e}");
                    break; // reconnect
                }
            }
            tokio::time::sleep(Duration::from_millis(POLL_INTERVAL_MS)).await;
        }
    }
}
```

import { useEffect } from 'react';
import { useThemeStore } from '@/stores/theme';

function applyTheme(mode: 'light' | 'dark' | 'auto') {
  let actual: 'light' | 'dark' = 'dark';
  if (mode === 'auto') {
    actual = window.matchMedia('(prefers-color-scheme: light)').matches ? 'light' : 'dark';
  } else {
    actual = mode;
  }
  document.documentElement.dataset.theme = actual;
}

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const mode = useThemeStore((s) => s.mode);
  useEffect(() => {
    applyTheme(mode);
    if (mode !== 'auto') return;
    const mq = window.matchMedia('(prefers-color-scheme: light)');
    const handler = () => applyTheme('auto');
    mq.addEventListener('change', handler);
    return () => mq.removeEventListener('change', handler);
  }, [mode]);
  return <>{children}</>;
}

import { create } from 'zustand';

export interface TelemetrySample {
  cpu_temp_c: number;
  sys_temp_c: number;
  cpu_fan_rpm: number;
  gpu_fan_rpm: number;
  timestamp_ms: number;
}

interface State {
  latest: TelemetrySample | null;
  history: TelemetrySample[];
  maxHistory: number;
  push: (s: TelemetrySample) => void;
  setMaxHistory: (n: number) => void;
}

export const useTelemetryStore = create<State>((set) => ({
  latest: null,
  history: [],
  maxHistory: 1800,
  push: (s) =>
    set((state) => {
      const history = [...state.history, s];
      if (history.length > state.maxHistory) history.shift();
      return { latest: s, history };
    }),
  setMaxHistory: (n) => set({ maxHistory: n }),
}));

import { useEffect } from 'react';
import { listen } from '@tauri-apps/api/event';
import { invoke } from '@tauri-apps/api/core';

export function useTelemetrySubscription() {
  const push = useTelemetryStore((s) => s.push);

  useEffect(() => {
    (async () => {
      const initial = await invoke<TelemetrySample | null>('get_telemetry');
      if (initial) push(initial);
    })();

    return () => { p.then((unl) => unl()); };

function Sparkline({ values, color }: { values: number[]; color: string }) {
  if (values.length < 2) return null;
  const min = Math.min(...values);
  const max = Math.max(...values);
  const range = max - min || 1;
  const points = values.map((v, i) => {
    const x = (i / (values.length - 1)) * 100;
    const y = 18 - ((v - min) / range) * 16;
    return `${x},${y}`;
  }).join(' ');
  return (
    <svg viewBox="0 0 100 20" preserveAspectRatio="none" className="w-full h-4 mt-1.5">
      <polyline fill="none" stroke={color} strokeWidth="1.2" points={points} />
    </svg>
  );
}

use serde::{Deserialize, Serialize};

#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct Point {
    pub temp_c: f32,
    pub pwm: u8,
}

#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct FanCurve {
    pub points: Vec<Point>,
}

impl FanCurve {
    pub fn pwm_at(&self, temp_c: f32) -> u8 {
        if self.points.is_empty() {
            return 0;
        }
        if temp_c <= self.points[0].temp_c {
            return self.points[0].pwm;
        }
        if temp_c >= self.points.last().unwrap().temp_c {
            return self.points.last().unwrap().pwm;
        }
        for w in self.points.windows(2) {
            let a = &w[0];
            let b = &w[1];
            if temp_c >= a.temp_c && temp_c <= b.temp_c {
                let span = b.temp_c - a.temp_c;
                let t = if span > 0.0 { (temp_c - a.temp_c) / span } else { 0.0 };
                let pwm_span = b.pwm as f32 - a.pwm as f32;
                return (a.pwm as f32 + pwm_span * t).round() as u8;
            }
        }
        self.points.last().unwrap().pwm
    }
}
```

    pub gpu_curve: RwLock<Option<FanCurve>>,
}

impl CurveRunner {
    pub fn new() -> Self {
        Self {
            cpu_curve: RwLock::new(None),
            gpu_curve: RwLock::new(None),
        }
    }

    pub fn set_cpu(&self, curve: Option<FanCurve>) { *self.cpu_curve.write() = curve; }
    pub fn set_gpu(&self, curve: Option<FanCurve>) { *self.gpu_curve.write() = curve; }
}

pub fn start(runner: Arc<CurveRunner>, app_state: Arc<AppState>) {
    tokio::spawn(async move {
        loop {
            tokio::time::sleep(Duration::from_millis(500)).await;
            let telem = app_state.last_telemetry.read().clone();
            let Some(t) = telem else { continue; };

            if let Some(curve) = runner.cpu_curve.read().clone() {
                let pwm = curve.pwm_at(t.cpu_temp_c);
                if let Ok(mut c) = IpcClient::connect() {
                    let _ = c.set_fan_mode_custom_cpu(pwm);
                }
            }
            if let Some(curve) = runner.gpu_curve.read().clone() {
                // GPU curve использует sys_temp_c как источник: оригинальный IpcRequest::GetTemperatures
                // возвращает только CPU и System temps. Корпусная температура коррелирует с GPU достаточно,
                // чтобы кривая работала на практике. Если в Phase 8 выяснится, что EC отдаёт отдельный GPU
                // temp — добавляем поле gpu_temp_c в Telemetry и переключаемся на него.
                let pwm = curve.pwm_at(t.sys_temp_c);
                if let Ok(mut c) = IpcClient::connect() {
                    let _ = c.set_fan_mode_custom_gpu(pwm);
                }
            }
        }
    });
}

export interface CurvePoint { temp_c: number; pwm: number; }

export function sortPoints(points: CurvePoint[]): CurvePoint[] {
  return [...points].sort((a, b) => a.temp_c - b.temp_c);
}

export function clampPoint(p: CurvePoint): CurvePoint {
  return {
    temp_c: Math.max(0, Math.min(100, p.temp_c)),
    pwm: Math.max(0, Math.min(100, p.pwm)),
  };
}

export function svgToTemp(x: number, width: number): number { return (x / width) * 100; }
export function svgToPwm(y: number, height: number): number { return 100 - (y / height) * 100; }
export function tempToSvg(temp: number, width: number): number { return (temp / 100) * width; }
export function pwmToSvg(pwm: number, height: number): number { return height - (pwm / 100) * height; }
```

    let show = MenuItem::with_id(app, "show", "Open window", true, None::<&str>)?;
    let quit = MenuItem::with_id(app, "quit", "Quit", true, None::<&str>)?;

            "fan_full" => { let _ = tauri::async_runtime::block_on(crate::tauri_cmds::set_fan_mode_internal("Cpu".into(), "Full".into(), None)); }

- [ ] **Step 1: handlers.rs — replace `.lock()` panic with try_lock + timeout**

```rust
use std::time::{Duration, Instant};

fn get_state_with_timeout(dur: Duration) -> Option<MutexGuard<'_, EcDevice>> {
    let deadline = Instant::now() + dur;
    loop {
        if let Some(g) = EC_DEVICE.try_lock() { return Some(g); }
        if Instant::now() >= deadline { return None; }
        std::thread::sleep(Duration::from_millis(10));
    }
}
```

Use this instead of `EC_DEVICE.lock().unwrap()` in handler entry points. If `None`, return `IpcResponse::Error("EC busy")`.
